}

// ============================================================
void reconnect() {
  while (!client.connected()) {
    Serial.print("Intentando conexión MQTT...");
    if (client.connect("ESP32Client")) {
      Serial.println("conectado");
    } else {
      Serial.print("falló, rc=");
      Serial.print(client.state());
      Serial.println(" reintentando en 5s");
      delay(5000);
    }
  }
}

/**
  getTemperatura(): Función que devuelve la temperatura (en grados Celsius) leída desde el pin del Termistor.
  OUTPUT:
    return (float): Temperatura en Grados Celsius (ºC).
*/
float getTemperatura() {
  int adcValue = analogRead(thermistorPin);

  if (adcValue <= 0 || adcValue >= 4095) {
    Serial.println("ADVERTENCIA: lectura ADC inválida");
    return -999.0;
  }

  double voltage = (double)adcValue / 4095.0 * 3.3;
  double Rt      = 10.0 * voltage / (3.3 - voltage);

  if (Rt <= 0) {
    Serial.println("ADVERTENCIA: Rt inválido");
    return -999.0;
  }

  double tempK = 1.0 / (1.0 / (273.15 + 25.0) + log(Rt / 10.0) / 3950.0);
  return (float)(tempK - 273.15);

}

/**

  INPUTS:

*/
void enviarAlertaTelegram(String mensaje) {
  if (WiFi.status() == WL_CONNECTED) {
    bool enviado = bot.sendMessage(CHAT_ID, mensaje, "");
    if (enviado) {
      Serial.println("Alerta Telegram enviada: " + mensaje);
    } else {
      Serial.println("Error al enviar alerta Telegram");
    }
  }
}
